<script>
  AOS.init({ duration: 1000, once: true });

  // Typing Effect
  const text = "ABHISHEK A K.";
  let i = 0;
  function type() {
    if (i < text.length) {
      document.getElementById("typing").innerHTML += text.charAt(i);
      i++;
      setTimeout(type, 100);
    }
  }
  type();

  // Parallax
  const parallax = document.getElementById('parallax');
  const layers = document.querySelectorAll('.layer');
  parallax.addEventListener('mousemove', (e) => {
    const x = e.clientX;
    const y = e.clientY;
    layers.forEach((layer, index) => {
      const speed = (index + 1) * 10;
      const xMove = (window.innerWidth / 2 - x) / speed;
      const yMove = (window.innerHeight / 2 - y) / speed;
      layer.style.transform = `translate(${xMove}px, ${yMove}px)`;
    });
  });

  // Cursor
  const cursor = document.getElementById('cursor');
  const links = document.querySelectorAll('.link');
  document.addEventListener('mousemove', (e) => {
    cursor.style.left = e.clientX + 'px';
    cursor.style.top = e.clientY + 'px';
  });
  links.forEach(link => {
    link.addEventListener('mouseenter', () => cursor.classList.add('cursor-grow'));
    link.addEventListener('mouseleave', () => cursor.classList.remove('cursor-grow'));
  });
</script>
